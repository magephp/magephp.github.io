37:      * @see \Mage\Command\BuiltIn\InstallCommand::run()

41:         $exitCode = 99;
42:         Console::output('Upgrading <bold>Magallanes</bold> ... ', 1, 0);
43: 
44:         $user = '';
45:         // Check if user is root
46:         Console::executeCommand('whoami', $user);
47:         $owner = posix_getpwuid(fileowner(__FILE__));
48:         $owner = $owner['name'];
49: 
50:         if ($user != 'root' && $user != $owner) {
51:             Console::output('<red>FAIL</red>', 0, 1);
52:             Console::output('You need to be the <bold>' . $owner . '</bold> user to perform the upgrade, or <bold>root</bold>.', 2);

54:             // Check version
55:             $version = json_decode(file_get_contents(self::UPGRADE));
56: 
57:             if ($version !== false && $version !== null) {
58:                 $versionCompare = version_compare(MAGALLANES_VERSION, $version->latest);
59:                 if ($versionCompare == 0) {
60:                     Console::output('<yellow>SKIP</yellow>', 0, 1);
61:                     Console::output('Your current version is up to date.', 2);
62:                     $exitCode = 0;
63:                 } elseif ($versionCompare == 1) {
64:                     Console::output('<yellow>SKIP</yellow>', 0, 1);
65:                     Console::output('Your current version is newer.', 2);
66:                     $exitCode = 0;
67:                 } elseif ($versionCompare == -1) {
68:                     // Download Package
69:                     $tarball = file_get_contents(str_replace('{version}', $version->latest, self::DOWNLOAD));
70:                     if ($tarball === false) {
71:                         Console::output('<red>FAIL</red>', 0, 1);
72:                         Console::output('Corrupted download.', 2);
73:                     } else {
74:                         $tarballFile = tempnam('/tmp', 'magallanes_download');
75:                         rename($tarballFile, $tarballFile . '.tar.gz');
76:                         $tarballFile .= '.tar.gz';
77:                         file_put_contents($tarballFile, $tarball);
78: 
79:                         Console::executeCommand('rm -rf ' . MAGALLANES_DIRECTORY);
80:                         Console::executeCommand('cd ' . dirname($tarballFile) . ' && tar xfz ' . $tarballFile);
81:                         Console::executeCommand('mv ' . dirname($tarballFile) . '/magallanes ' . MAGALLANES_DIRECTORY);
82: 
83:                         Console::output('<green>OK</green>', 0, 1);
84:                         $exitCode = 0;
85:                     }
86:                 } else {
