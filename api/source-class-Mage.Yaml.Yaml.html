 27:     /**
 28:      * Parses YAML into a PHP array.
 29:      *
 30:      * The parse method, when supplied with a YAML stream (string or file),
 31:      * will do its best to convert YAML in a file into a PHP array.
 32:      *
 33:      *  Usage:
 34:      *  <code>
 35:      *   $array = Yaml::parse('config.yml');
 36:      *   print_r($array);
 37:      *  </code>
 38:      *
 39:      * As this method accepts both plain strings and file names as an input,
 40:      * you must validate the input before calling this method. Passing a file
 41:      * as an input is a deprecated feature and will be removed in 3.0.
 42:      *
 43:      * @param string $input Path to a YAML file or a string containing YAML
 44:      * @param bool $exceptionOnInvalidType True if an exception must be thrown on invalid types false otherwise
 45:      * @param bool $objectSupport True if object support is enabled, false otherwise
 46:      *
 47:      * @return array The YAML converted to a PHP array
 48:      *
 49:      * @throws ParseException If the YAML is not valid
 50:      *
 51:      * @api
 52:      */
 53:     public static function parse($input, $exceptionOnInvalidType = false, $objectSupport = false)
 54:     {
 55:         // if input is a file, process it
 56:         $file = '';
 57:         if (strpos($input, "\n") === false && is_file($input)) {
 58:             if (false === is_readable($input)) {
 59:                 throw new ParseException(sprintf('Unable to parse "%s" as the file is not readable.', $input));
 60:             }
 61: 
 62:             $file  = $input;
 63:             $input = file_get_contents($file);
 64:         }
 65: 
 66:         $yaml = new Parser();
 67: 
 68:         try {
 69:             return $yaml->parse($input, $exceptionOnInvalidType, $objectSupport);
 70:         } catch (ParseException $e) {
 71:             if ($file) {
 72:                 $e->setParsedFile($file);
 73:             }
 74: 
 75:             throw $e;
 76:         }
 77:     }
 78: 
 79:     /**
 80:      * Dumps a PHP array to a YAML string.
 81:      *
 82:      * The dump method, when supplied with an array, will do its best
 83:      * to convert the array into friendly YAML.
 84:      *
 85:      * @param array $array PHP array
 86:      * @param int $inline The level where you switch to inline YAML
 87:      * @param int $indent The amount of spaces to use for indentation of nested nodes.
 88:      * @param bool $exceptionOnInvalidType true if an exception must be thrown on invalid types (a PHP resource or object), false otherwise
 89:      * @param bool $objectSupport true if object support is enabled, false otherwise
 90:      *
 91:      * @return string A YAML string representing the original PHP array
 92:      *
 93:      * @api
 94:      */
 95:     public static function dump($array, $inline = 2, $indent = 4, $exceptionOnInvalidType = false, $objectSupport = false)
 96:     {
 97:         $yaml = new Dumper();

 99: 
100:         return $yaml->dump($array, $inline, 0, $exceptionOnInvalidType, $objectSupport);
101:     }
