 73:     /**
 74:      * Runns a Magallanes Command
 75:      * @throws Exception
 76:      */
 77:     public function run($arguments)
 78:     {
 79:         $exitCode = 10;
 80: 
 81:         // Declare a Shutdown Closure
 82:         register_shutdown_function(function () {
 83:             // Only Unlock if there was an error
 84:             if (error_get_last() !== null) {
 85:                 if (file_exists(getcwd() . '/.mage/~working.lock')) {
 86:                     unlink(getcwd() . '/.mage/~working.lock');
 87:                 }
 88:             }
 89:         });
 90: 
 91:         $config = self::$config = new Config;
 92:         $configError = false;
 93:         try {
 94:             // Load configuration
 95:             $config->load($arguments);
 96:         } catch (Exception $exception) {
 97:             $configError = $exception->getMessage();
 98:         }
 99: 
100:         // Command Option
101:         $commandName = $config->getArgument(0);
102: 
103:         // Logging
104:         $showGreetings = true;
105: 
106:         if (in_array($commandName, self::$paramsNotRequiringEnvironment)) {
107:             self::$logEnabled = false;
108:             $showGreetings = false;
109:         } else {
110:             self::$logEnabled = $config->general('logging', false);
111:         }
112: 
113:         self::$verboseLogEnabled = self::isVerboseLoggingEnabled();
114: 
115:         // Greetings
116:         if ($showGreetings) {
117:             if (!self::$logEnabled) {
118:                 self::output('Starting <blue>Magallanes</blue>', 0, 2);
119:             } else {
120:                 self::output('Starting <blue>Magallanes</blue>', 0, 1);
121:                 self::log("Logging enabled");
122:                 self::output('<bold>Logging enabled:</bold> <purple>' . self::getLogFile() . '</purple>', 1, 2);
123:             }
124:         }
125: 
126:         // Run Command - Check if there is a Configuration Error
127:         if ($configError !== false) {
128:             self::output('<red>' . $configError . '</red>', 1, 2);
129:         } else {
130:             // Run Command and check for Command Requirements
131:             try {
132:                 $command = Factory::get($commandName, $config);
133: 
134:                 if ($command instanceof RequiresEnvironment) {
135:                     if ($config->getEnvironment() === false) {
136:                         throw new Exception('You must specify an environment for this command.');
137:                     }
138:                 }
139: 
140:                 // Run the Command
141:                 $exitCode = $command->run();
142: 
143:                 // Check for errors
144:                 if (is_int($exitCode) && $exitCode !== 0) {
145:                     throw new Exception('Command execution failed with following exit code: ' . $exitCode, $exitCode);
146:                 } elseif (is_bool($exitCode) && !$exitCode) {
147:                     $exitCode = 1;
148:                     throw new Exception('Command execution failed.', $exitCode);
149:                 }
150:             } catch (Exception $exception) {
151:                 self::output('<red>' . $exception->getMessage() . '</red>', 1, 2);
152:             }
153:         }
154: 
155:         if ($showGreetings) {
156:             self::output('Finished <blue>Magallanes</blue>', 0, 2);
157:             if (file_exists(getcwd() . '/.mage/~working.lock')) {
158:                 unlink(getcwd() . '/.mage/~working.lock');
159:             }
160:         }
161: 
162:         // Check if logs need to be deleted
163:         self::checkLogs($config);
164: 
165:         return $exitCode;
166:     }

275:     private static function checkLogs(Config $config)
276:     {
277:         if (self::$logEnabled) {
278:             $maxLogs = $config->general('maxlogs', 30);
279: 
280:             $logs = array();
281:             foreach (new RecursiveDirectoryIterator(getcwd() . '/.mage/logs', RecursiveDirectoryIterator::SKIP_DOTS) as $log) {
282:                 /* @var $log SplFileInfo */
283:                 if (strpos($log->getFilename(), 'log-') === 0) {
284:                     $logs[] = $log->getFilename();
285:                 }
286:             }
287: 
288:             sort($logs);
289:             if (count($logs) > $maxLogs) {
290:                 $logsToDelete = array_slice($logs, 0, count($logs) - $maxLogs);
291:                 foreach ($logsToDelete as $logToDeelte) {
292:                     unlink(getcwd() . '/.mage/logs/' . $logToDeelte);
293:                 }
294:             }
295:         }
296:     }
